const express = require('express'); const https = require('https'); const fs = require('fs'); const app = express(); // Load your SSL certificate const options = { key: fs.readFileSync('key.pem'), cert: fs.readFileSync('cert.pem') }; // Route to serve a YouTube video app.get('/video', (req, res) => { const videoId = req.query.id; // Get video ID from query params if (videoId) { const videoUrl = `https://www.youtube.com/embed/${videoId}`; res.redirect(videoUrl); // Redirect to YouTube embed } else { res.status(400).send('Video ID is required'); } }); // Start HTTPS server https.createServer(options, app).listen(3000, () => { console.log('Server running on https://localhost:3000'); });
